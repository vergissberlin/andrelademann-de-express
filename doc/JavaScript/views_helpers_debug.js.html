	Handlebars = require('handlebars');

Handlebars.registerHelper('debug', function (optionalValue) {
	console.info('\nCurrent Context');
	console.info('====================');
	console.log(this);

	if (optionalValue) {
		console.info('Value');
		console.info('====================');
		console.log(optionalValue);
	}
});
